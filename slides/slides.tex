\begin{frame}{Q1 -- Bellman Expectation: $\Vpi$ and $\Qpi$}
  \textbf{State-value function under policy $\pi$:}
  \[
    \boxed{\Vpi(s) = \sum_a \pi(a|s)\sum_{s'} P(s'|s,a)\bigl[R(s,a,s') + \gamma\,\Vpi(s')\bigr]}
  \]

  \textbf{Action-value (Q) function:}
  \[
    \boxed{\Qpi(s,a) = \sum_{s'} P(s'|s,a)\bigl[R(s,a,s') + \gamma\,\Vpi(s')\bigr]}
  \]

  \textbf{Consistency relation:}
  \[
    \Vpi(s) = \sum_a \pi(a|s)\,\Qpi(s,a)
  \]
\end{frame}

% в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ
\begin{frame}{Q1 -- Bellman Expectation: Iterative Evaluation}
  \begin{block}{Iterative Policy Evaluation}
    Start from $V_0 = \mathbf{0}$, repeat until $\|V_{k+1}-V_k\|_\infty < \theta$:
    \[
      V_{k+1}(s) = \sum_a \pi(a|s)\sum_{s'} P(s'|s,a)\bigl[R + \gamma\,V_k(s')\bigr]
    \]
  \end{block}

  \medskip
  \textbf{Linear system} (uniform policy, $\gamma=1$, $V(\text{Sleep})=0$):
  \small
  \begin{align*}
    V(C_1) &= \tfrac{1}{2}[-2+V(C_2)] + \tfrac{1}{2}[-1+V(\text{FB})]\\
    V(C_2) &= \tfrac{1}{2}[-2+V(C_3)] + \tfrac{1}{2}[0+V(\text{Sleep})]\\
    V(C_3) &= \tfrac{1}{2}[10+V(\text{Sleep})] + \tfrac{1}{2}[1+0.2\,V(C_1)+0.4\,V(C_2)+0.4\,V(C_3)]\\
    V(\text{FB}) &= \tfrac{1}{2}[-1+V(\text{FB})] + \tfrac{1}{2}[0+V(C_1)]
  \end{align*}
\end{frame}

\begin{frame}{Q2 -- Hand Computation: $k=1$ and $k=2$}
  \begin{columns}[T]
    \column{0.48\textwidth}
    \begin{block}{$k=1$: from $V_0=0$}
      All 4 actions give $R=-1$, $V_0(s')=0$:
      \[
        V_1(s) = \tfrac{1}{4}(-1\times 4) = -1.0
        \quad\forall s\notin\{0,15\}
      \]
    \end{block}

    \column{0.48\textwidth}
    \begin{block}{$k=2$: State 1 (top row, col~1)}
      \scriptsize
      N$\to$1 (wall): $-1+(-1)=-2$\\
      S$\to$5: $-1+(-1)=-2$\\
      W$\to$0 (term): $-1+0=-1$\\
      E$\to$2: $-1+(-1)=-2$\\[4pt]
      \[
        V_2(1) = \tfrac{1}{4}(-2-2-1-2) = \mathbf{-1.75}
      \]
      \[
        V_2(5) = \tfrac{1}{4}(-2\times4) = \mathbf{-2.00}
      \]
    \end{block}
  \end{columns}

  \medskip
  \begin{alertblock}{Key detail}
    N from row~0 \emph{reflects} back to same cell (wall), not wrapping to terminal~0.
  \end{alertblock}
\end{frame}

\begin{frame}{Q2 -- Proper Policy \& Convergence for $\gamma=1$}
  \begin{itemize}
    \item \textbf{Proper Policy:} A policy where the probability of reaching a terminal state is 1 from any starting state.
    \item \textbf{Theorem:} If the policy is proper, the iterative process converges to a unique, finite fixed point $V^\pi$ even for $\gamma=1$.
    \item \textbf{Mechanism:} Terminal states act as "leaks" that prevent values from diverging to $-\infty$.
    \item \textbf{Spectral Radius:} The effective transition operator is substochastic with $\rho(P) < 1$.
  \end{itemize}
\end{frame}

% в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ

  \begin{columns}[T]
    \column{0.48\textwidth}
    We can group the 14 non-terminal states into \textbf{5 classes}:
    \begin{itemize}
      \item \textbf{A}: Adjacent to terminal
      \item \textbf{B}: Edge middle
      \item \textbf{C}: Far corners
      \item \textbf{D}: Near-terminal interior

    \end{itemize}

    \column{0.48\textwidth}

      \item $B_{k+1} = -1 + \frac{1}{4}(A_k + B_k + C_k + E_k)$
      \item $C_{k+1} = -1 + \frac{1}{2}(B_k + C_k)$
      \item $D_{k+1} = -1 + \frac{1}{4}(A_k + A_k + E_k + D_k)$
      \item $E_{k+1} = -1 + \frac{1}{4}(B_k + B_k + D_k + E_k)$
    \end{itemize}
  \end{columns}
\end{frame}

\section{Q3: Jack's Car Rental}
% ============================================================

\begin{frame}{Q3 -- Jack's Car Rental: MDP (Lecture 3, pp.~14--15)}
  \begin{columns}[T]
    \column{0.50\textwidth}
    \textbf{Two locations; rent/return $\sim$ Poisson:}\\[4pt]

    \begin{tabular}{ll}
      \toprule
      Parameter & Value \\
      \midrule
      Max cars / loc & 20 \\
      Max moves / night & 5 \\
      Rental reward & $+\$10$ \\
      Move cost & $\$2$/car \\
      $\lambda_{\text{rent1}},\lambda_{\text{rent2}}$ & 3,\ 4 \\
      $\lambda_{\text{ret1}},\lambda_{\text{ret2}}$ & 3,\ 2 \\
      $\gamma$ & 0.9 \\
      \bottomrule
    \end{tabular}

    \column{0.46\textwidth}
    \begin{block}{State \& Action}
      $s=(n_1,n_2)$: cars at start of day\\
      $a\in[-5,5]$: move loc2$\to$loc1
    \end{block}

    \begin{block}{Bellman Equation}
      \scriptsize
      \[
        V(s) = {-2|a|}+\E\!\bigl[10\cdot r + \gamma V(s')\bigr]
      \]
    \end{block}

    \begin{alertblock}{Speed-up}
      \scriptsize
      Precompute $\text{EXP\_REW}[n]$, $\text{TRANS}[n,n']$\\
      $\Rightarrow$ matrix-vector product per state
    \end{alertblock}
  \end{columns}
\end{frame}

\begin{frame}{Q3 -- Precomputed Dynamics \& Policy Iteration}
  \begin{columns}[T]
    \column{0.52\textwidth}
    \textbf{Precomputed (per location):}
    \[
      \text{EXP}[n] = \sum_{k=0}^{11} P(k;\lambda)\cdot\min(k,n)\cdot 10
    \]
    \[
      \text{TRANS}[n,n'] = \sum_{\text{req,ret}} P_r P_d\,\mathbf{1}[n'=\cdots]
    \]

    \textbf{Future value (matrix product):}
    \[
      \E[V(n_1',n_2')] = T_1[n_1,:]\cdot V\cdot T_2[n_2,:]^\top
    \]

    \column{0.44\textwidth}
    \begin{block}{Policy Iteration}
      \textbf{1.} Evaluate $V^{\pi_k}$\\
      \quad (Bellman until $\|\Delta V\|<\theta$)\\[4pt]
      \textbf{2.} Improve\\

      \textbf{3.} Repeat until $\pi_{k+1}=\pi_k$\\[4pt]
      \textbf{Result:} converges in $\approx\!4$ steps
    \end{block}

    \begin{block}{Interpretation}

\begin{frame}{Q3 -- Optimal Policy: Insights \& Patterns}
  \begin{itemize}
    \item \textbf{Inventory Equilibrium:} A diagonal region ($n_1 \approx n_2$) where $a=0$ is optimal. Moving cars incurs cost with little marginal benefit.
    \item \textbf{Demand-Driven Redistribution:} Loc~2 has higher demand ($\lambda{=}4$ vs $3$). The policy aggressively moves cars from Loc~1 $\to$ Loc~2.
    \item \textbf{Saturation Effect:} When a location is at capacity (20 cars), sending more cars there yields diminishing returns.
    \item \textbf{Zig-Zag Boundary:} Occurs due to:
    \begin{enumerate}
      \item \textbf{Discrete Action Space:} $a \in [-5, 5]$ jumps in integer steps.
      \item \textbf{Non-linear Demand:} Poisson cumulative rewards shift boundaries abruptly.
      \item \textbf{Cost Trade-off:} Small state shifts flip the balance between $\$2$ move cost and expected rental revenue.
    \end{enumerate}
  \end{itemize}
\end{frame}

\section{Q4: Contraction Mapping Theorem}
% ============================================================

% в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ
\begin{frame}{Q4 -- Banach Theorem: Existence \& Uniqueness}
  \begin{theorem}[Banach Fixed-Point]
    Let $(X, d)$ be a complete metric space. If $T:X \to X$ is a $\gamma$-contraction, then $T$ has a \textbf{unique fixed point} $x^*$.
  \end{theorem}
  \textbf{Proof Sketch (Existence):}
  \begin{enumerate}
    \item Define sequence $x_{k+1} = T x_k$.
    \item Show it is a \textbf{Cauchy Sequence}:
    \[
      d(x_{k+1}, x_k) \leq \gamma^k d(x_1, x_0) \implies \sum_{i=n}^m d(x_{i+1}, x_i) \text{ converges.}
    \]
    \item Since $X$ is complete, $x_k \to x^*$. By continuity, $T x^* = x^*$.
  \end{enumerate}
\end{frame}

% в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ
\begin{frame}{Q4 -- Proof: Uniqueness \& Intuition}
  \textbf{Proof (Uniqueness):}
  Suppose $x^*$ and $y^*$ are both fixed points. Then:
  \[
    d(x^*, y^*) = d(T x^*, T y^*) \leq \gamma \cdot d(x^*, y^*)
  \]
  Since $\gamma < 1$, the only solution is $d(x^*, y^*) = 0 \implies x^* = y^*$.

  \medskip
  \begin{block}{Intuition: Error Shrinking Mechanism}
    The Bellman update acts as an error-shrinking mechanism:
    \[
      \text{Error}_{k+1} \leq \gamma \cdot \text{Error}_{k}
    \]
    After $k$ steps, the initial error is reduced by $\gamma^k$. As $k \to \infty$, the value function is "forced" into the unique fixed point.
  \end{block}
\end{frame}

% в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ
\begin{frame}{Q4 -- $\Tpi$ is a $\gamma$-Contraction}
  Define: $(\Tpi V)(s) = \sum_a\pi(a|s)\sum_{s'}P(s'|s,a)\bigl[R+\gamma V(s')\bigr]$

  \begin{proof}
    \vspace{-4pt}
    \begin{align*}
      |(\Tpi V_1)(s) - (\Tpi V_2)(s)|
      &= \gamma\Bigl|\sum_a\pi(a|s)\sum_{s'}P(s'|s,a)[V_1(s')-V_2(s')]\Bigr|\\
      &\leq \gamma\,\norm{V_1-V_2}\underbrace{\textstyle\sum_a\pi(a|s)\sum_{s'}P(s'|s,a)}_{=1}
    \end{align*}
    Take $\max_s$: $\;\norm{\Tpi V_1-\Tpi V_2}\leq\gamma\,\norm{V_1-V_2}.$
  \end{proof}

  \begin{alertblock}{Consequence}
    Iterative policy evaluation converges to the unique $V^\pi$ from \emph{any} $V_0$.
  \end{alertblock}
\end{frame}

% в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ
\begin{frame}{Q4 -- $\Tstar$ is a $\gamma$-Contraction}
  Define: $(\Tstar V)(s) = \max_a\sum_{s'}P(s'|s,a)\bigl[R+\gamma V(s')\bigr]$

  \begin{proof}
    Using $|\max f - \max g|\leq\max|f-g|$:
    \[
      \norm{\Tstar V_1-\Tstar V_2}\leq\gamma\,\norm{V_1-V_2}
    \]
  \end{proof}

  \begin{columns}[T]
    \column{0.5\textwidth}
    \begin{block}{Value Iteration Bound}
      \[
        \norm{V_k-\Vstar}\leq\gamma^k\,\norm{V_0-\Vstar}
      \]
    \end{block}

    \column{0.46\textwidth}
    \begin{block}{Numerical Verification}
      \scriptsize
      For $\gamma\in\{0.5,0.9,0.99\}$, 500 random pairs:
      \[
        \frac{\norm{TV_1-TV_2}}{\norm{V_1-V_2}}\leq\gamma+10^{-9}\ \checkmark
      \]
    \end{block}
  \end{columns}
\end{frame}

% в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ
\begin{frame}{Q4 -- Policy Iteration Convergence}
  \begin{theorem}[Policy Iteration Convergence]
    Policy iteration converges to $\pistar$ in finitely many steps.
  \end{theorem}

  \begin{proof}[Sketch]
    \begin{enumerate}
      \item \textbf{Monotone improvement} (greedy step always $\geq$ current policy):
      \[
        V^{\pi'}(s)\geq Q^\pi(s,\pi'(s))=\max_a Q^\pi(s,a)\geq V^\pi(s)
      \]
      \item \textbf{Finite} policies: at most $|\mathcal{A}|^{|S|}$ deterministic policies.
      \item \textbf{Monotone~+~Finite $\Rightarrow$ terminates.}
        At termination $\pi^{k+1}=\pi^k$, so $V^{\pi^k}$ satisfies the Bellman\\
        Optimality Equation $\Rightarrow$ $V^{\pi^k}=V^*$.
    \end{enumerate}
  \end{proof}
\end{frame}
